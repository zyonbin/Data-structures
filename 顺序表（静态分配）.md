# 顺序表（静态分配）
## 一、顺序表结构体类型定义
```c
#define ListSize 100  // 最大允许长度
typedef struct {
    ListData data[ListSize];  // 存储元素数组空间
    int length;                // 当前有效元素个数
} SqList;
```
## 二、全部操作函数声明
1. `void InitList(SqList &L);` 初始化空表
2. `void DestroyList(SqList &L);` 销毁表（静态分配仅重置长度）
3. `void ClearList(SqList &L);` 清空表
4. `int ListEmpty(SqList L);` 判断空表，空返回1，否则0
5. `int ListLength(SqList L);` 获取当前有效长度
6. `ListData GetElem(SqList L, int i);` 获取第i位元素（1起始）
7. `int LocateElem(SqList L, ListData x);` 查找x，返回位序，无返回-1
8. `int IsIn(SqList L, ListData x);` 判断x是否存在，存在返回1
9. `int NextElem(SqList L, ListData x);` 获取x后继位序
10. `int PriorElem(SqList L, ListData x);` 获取x前驱位序
11. `int ListInsert(SqList &L, int i, ListData x);` 第i位插入x，成功1失败0
12. `int ListDelete(SqList &L, int i);` 删除第i位元素，成功1失败0
## 三、基础操作函数
### 1. InitList(&L) 初始化线性表L为空表
```c
void InitList(SqList &L) {
    L.length = 0;
}
```

### 2. DestroyList(&L) 回收销毁线性表L
```c
void DestroyList(SqList &L) {
    L.length = 0;
}
```

### 3. ClearList(&L) 清空线性表L
```c
void ClearList(SqList &L) {
    L.length = 0;
}
```

### 4. ListEmpty(L) 判断线性表L是否为空
为空返回1，非空返回0
写法1：
```c
int ListEmpty(SqList L) {
    return !(L.length);
}
```
写法2：
```c
int ListEmpty(SqList L) {
    if (L.length == 0) return 1;
    else return 0;
}
```

### 5. ListLength(L) 求线性表L当前有效长度
```c
int ListLength(SqList L) {
    return L.length;
}
```

### 6. GetElem(L, i) 获取线性表L第i个元素(1-based)
i 取值范围 `1 ≤ i ≤ L.length`，非法返回错误标识
```c
ListData GetElem(SqList L, int i) {
    if (i >= 1  && i <= L.length)
        return L.data[i-1];
    else 
        return "Error";
}
```

## 四、查找类函数
### 7-1. LocateElem(L, x) 查找元素x，返回位序（1-based），无则返回-1
while循环版本：
```c
int LocateElem(SqList L, ListData x) {
    int i = 0;
    while ((i < L.length) && (L.data[i] != x)) i++;
    if (i < L.length) 
        return i + 1;  // 数组下标转逻辑位序:位序=下标+1
    else 
        return -1;
}
```
for循环版本：
```c
int LocateElem(SqList L, ListData x) {
    for (int i = 0; i < L.length; i++)
        if (L.data[i] == x) return i + 1;
    return -1;
}
```

### 7-2. IsIn(L, x) 判断x是否存在表中
存在返回1，不存在返回0
while循环版本：
```c
int IsIn(SqList L, ListData x) {
    int i = 0, found = 0;
    while ((i < L.length) && !found) {
        if (L.data[i] == x)
            found = 1;
        else 
            i++;
    }
    return found;
}
```
for循环版本：
```c
int IsIn(SqList L, ListData x) {
    int found = 0;
    for (int i = 0; i < L.length; i++) {
        if (L.data[i] == x) {
            found = 1;
            break; // 找到直接跳出循环
        }
    }
    return found;
}
```
### 8. NextElem(L, x) 求x的直接后继位序
找到x后有后继返回后继位序，无x/无后继返回-1
x下标\(i \in [0, L.length-1)\)->后继下标\(i+1\)
x位序\(i+1 \in [1, L.length)\)->后继位序\(i+2\)
```c
int NextElem(SqList L, ListData x) {
    int i = 0;
    while ((i < L.length) && (L.data[i] != x)) i++;
    // if ((i >= 0) && (i < L.length - 1)) 
    if (i < L.length - 1) 
        return i + 2;//求位序
        // return i + 1;//求下标
    else 
        return -1;
}
```

### 9. PriorElem(L, x) 求x的直接前驱位序
找到x后有前驱返回前驱位序，无x/无前驱返回-1
x下标\(i \in [1, L.length)\)->前驱下标\(i - 1\)
x位序\(i+1 \in [2, L.length]\)->前驱位序\(i\)
```c
int PriorElem(SqList L, ListData x) {
    int i = 0;
    while ((i < L.length) && (L.data[i] != x)) i++;
    if ((i > 0) && (i < L.length)) 
        return i;//求位序
        // return i - 1;//求下标
    else 
        return -1;
}
```

## 五、增删操作函数
### 10. ListInsert(&L, i, x) 在第i个元素前插入x
插入成功返回1，失败（i非法/表满）返回0
#### 写法1：先移动元素，再更新长度
```c
int ListInsert(SqList &L, int i, ListData x) {
    // i合法区间 [1, L.length+1]，或数组已满
    if ((i < 1) || (i > L.length + 1) || (L.length == ListSize))
        return 0;
    else {
        // 从后往前批量后移元素，腾出i-1下标位置
        for (int j = L.length; j > i - 1; j--)
            L.data[j] = L.data[j - 1];
        L.data[i - 1] = x;
        L.length++;
        return 1;
    }
}
```
$L.data[x] = L.data[x-1]$
$x-1 \in [i-1,\ L.length-1] \implies x \in [i,\ L.length]$
循环变量 j 的合法取值区间：
① $x = j,\quad j \in [i,\ L.length],\quad L.data[j] = L.data[j-1]$
变量代换规则：
② $x = j+a,\quad j \in [i-a,\ L.length-a],\quad L.data[j+a] = L.data[j+a-1]$
a = 1的变量代换：
```c
int ListInsert(SqList &L, int i, ListData x) {
    if ((i < 1) || (i > L.length + 1) || (L.length == ListSize))
        return 0;
    else {
        // a=1 代换实现：j 区间 [i-1, L.length-1]
        for (int j = L.length - 1; j >= i - 1; j--)
            L.data[j + 1] = L.data[j];  
        L.data[i - 1] = x;
        L.length++;
        return 1;
    }
}
```
#### 写法2：先更新长度，再移动元素
```c
int ListInsert(SqList &L, int i, ListData x) {
    if ((i < 1) || (i > L.length + 1) || (L.length == ListSize))
        return 0;
    else {
        L.length++;
        for (int j = L.length - 1; j > i - 1; j--)
            L.data[j] = L.data[j - 1];
        L.data[i - 1] = x;
        return 1;
    }
}
```
$L.data[x] = L.data[x-1]$
原 
$x-1 \in [i-1,\ L.length-1]$
$\quad\quad\quad\quad\quad\quad\quad\quad\downarrow L.length++$
$x-1 \in [i-1,\ L.length-2]\implies x \in [i,\ L.length-1]$
循环变量 j 的合法取值区间：
① $x = j,\ j \in [i,\ L.length-1]$ 
变量代换规则：
② $x = j+a,\ j \in [i-a,\ L.length-a-1]$
a = 1的变量代换：
```c
int ListInsert(SqList &L, int i, ListData x) {
    if ((i < 1) || (i > L.length + 1) || (L.length == ListSize))
        return 0;
    else {
        L.length++;
        for (int j = L.length - 2; j >= i - 1; j--)
            L.data[j + 1] = L.data[j];
        L.data[i - 1] = x;
        return 1;
    }
}
```
### 11. ListDelete(&L, i) 删除第i个位置元素
删除成功返回1，失败（i非法）返回0
#### 写法1：先移动元素，再更新长度
```c
int ListDelete(SqList &L, int i) {
    // i合法区间 [1, L.length]
    if ((i < 1) || (i > L.length))
        return 0;
    // 从前往后批量前移元素，覆盖i-1下标位置
    for (int j = i - 1; j <= L.length - 2; j++)
        L.data[j] = L.data[j + 1];
    --L.length;
    return 1;
}
```
$L.data[x] = L.data[x+1]$
$x+1 \in [i,\ L.length-1] \implies x \in [i-1,\ L.length-2]$
循环变量 j 的合法取值区间：
① $x = j,\quad j \in [i-1,\ L.length-2],\quad L.data[j] = L.data[j+1]$
变量代换规则：
② $x = j+a,\quad j \in [i-a-1,\ L.length-2-a],\quad L.data[j+a] = L.data[j+a+1]$
a = -1的变量代换：
```c
int ListDelete(SqList &L, int i) {
    if ((i < 1) || (i > L.length))
        return 0;
    // a=-1 代换实现：j 区间 [i, L.length-1]
    for (int j = i; j <= L.length - 1; j++)
        L.data[j - 1] = L.data[j];  
    --L.length;
    return 1;
}
```

#### 写法2：先更新长度，再移动元素
```c
int ListDelete(SqList &L, int i) {
    if ((i < 1) || (i > L.length))
        return 0;
    L.length--;
    for (int j = i - 1; j <= L.length - 1; j++)
        L.data[j] = L.data[j + 1];
    return 1;
}
```
$L.data[x] = L.data[x+1]$
原 
$x+1 \in [i,\ L.length-1]$
$\quad\quad\quad\quad\quad\quad\quad\quad\downarrow L.length--$
$x+1 \in [i,\ L.length]\implies x \in [i-1,\ L.length-1]$
循环变量 j 的合法取值区间：
① $x = j,\ j \in [i-1,\ L.length-1]$ 
变量代换规则：
② $x = j+a,\ j \in [i-a-1,\ L.length-a-1]$
a = -1的变量代换：
```c
int ListDelete(SqList &L, int i) {
    if ((i < 1) || (i > L.length))
        return 0;
    L.length--;
    for (int j = i; j <= L.length; j++)
        L.data[j - 1] = L.data[j];
    return 1;
}
```
## 六、C 语言实现
```c
#include <stdio.h>

#define ListSize 100
typedef int ListData;

// 顺序表结构体
typedef struct {
    ListData data[ListSize];
    int length;
} SqList;

// 1. 初始化空表
void InitList(SqList *L) {
    L->length = 0;
}

// 2. 销毁顺序表
void DestroyList(SqList *L) {
    L->length = 0;
}

// 3. 清空顺序表
void ClearList(SqList *L) {
    L->length = 0;
}

// 4. 判断是否为空
int ListEmpty(SqList L) {
    return !(L.length);
}

// 5. 获取表长度
int ListLength(SqList L) {
    return L.length;
}

// 6. 获取第i个元素，成功返回1，元素存入e；失败返回0
int GetElem(SqList L, int i, ListData *e) {
    if (i >= 1 && i <= L.length) {
        *e = L.data[i - 1];
        return 1;
    }
    return 0;
}

// 7-1 查找元素，返回位序，无则-1
int LocateElem(SqList L, ListData x) {
    for (int i = 0; i < L.length; i++) {
        if (L.data[i] == x)
            return i + 1;
    }
    return -1;
}

// 7-2 判断元素是否存在
int IsIn(SqList L, ListData x) {
    for (int i = 0; i < L.length; i++) {
        if (L.data[i] == x)
            return 1;
    }
    return 0;
}

// 8 获取x直接后继位序
int NextElem(SqList L, ListData x) {
    int i = 0;
    while (i < L.length && L.data[i] != x)
        i++;
    if (i < L.length - 1)
        return i + 2;
    return -1;
}

// 9 获取x直接前驱位序
int PriorElem(SqList L, ListData x) {
    int i = 0;
    while (i < L.length && L.data[i] != x)
        i++;
    if (i > 0 && i < L.length)
        return i;
    return -1;
}

// 10 插入元素
int ListInsert(SqList *L, int i, ListData x) {
    if (i < 1 || i > L->length + 1 || L->length == ListSize)
        return 0;
    // 元素后移
    for (int j = L->length; j > i - 1; j--)
        L->data[j] = L->data[j - 1];
    L->data[i - 1] = x;
    L->length++;
    return 1;
}

// 11 删除元素
int ListDelete(SqList *L, int i) {
    if (i < 1 || i > L->length)
        return 0;
    // 元素前移覆盖
    for (int j = i - 1; j <= L->length - 2; j++)
        L->data[j] = L->data[j + 1];
    L->length--;
    return 1;
}

// 辅助打印函数
void PrintList(SqList L) {
    printf("顺序表(长度=%d): ", L.length);
    for (int i = 0; i < L.length; i++)
        printf("%d ", L.data[i]);
    printf("\n");
}

// 测试主函数
int main() {
    SqList L;
    ListData val;

    // 初始化
    InitList(&L);
    printf("【初始化】");
    PrintList(L);
    printf("是否为空：%s\n\n", ListEmpty(L) ? "是" : "否");

    // 插入测试
    ListInsert(&L, 1, 10);
    ListInsert(&L, 2, 20);
    ListInsert(&L, 3, 30);
    ListInsert(&L, 2, 15);
    printf("【插入10,20,30,第2位插入15】");
    PrintList(L);

    // 取值、查找、前驱后继
    GetElem(L, 3, &val);
    printf("第3位元素：%d\n", val);
    printf("20的位序：%d\n", LocateElem(L, 20));
    printf("99是否存在：%s\n", IsIn(L, 99) ? "存在" : "不存在");
    printf("20前驱位序：%d\n", PriorElem(L, 20));
    printf("20后继位序：%d\n\n", NextElem(L, 20));

    // 删除
    ListDelete(&L, 2);
    printf("【删除第2位】");
    PrintList(L);

    // 清空销毁
    ClearList(&L);
    printf("【清空表】");
    PrintList(L);
    DestroyList(&L);

    return 0;
}
```
