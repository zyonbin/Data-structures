# 顺序表（动态分配）
## 一、顺序表结构体与宏定义
```c
#define LIST_INIT_SIZE 100    // 初始分配容量
#define LISTINCREMENT 10      // 扩容增量
typedef int ListData;         // 数据元素类型，可按需替换

// 动态分配顺序表结构体
typedef struct {
    ListData* data;         // 动态数组基指针
    int length;             // 当前有效元素个数
    int Listsize;           // 整体分配存储容量
} SeqList;

// 状态码定义
#define OK 1
#define ERROR 0
#define OVERFLOW -2
```

## 二、全部操作函数声明
1. `int InitList(SeqList &L);` 初始化空表，分配堆内存
2. `int DestroyList(SeqList &L);` 销毁表，释放堆内存
3. `int ClearList(SeqList &L);` 清空表（不释放内存）
4. `int ListEmpty(SeqList L);` 判断空表，空返回1，否则0
5. `int ListLength(SeqList L);` 获取当前有效长度
6. `ListData GetElem(SeqList L, int i);` 获取第i位元素（1起始）
7. `int LocateElem(SeqList L, ListData x);` 查找x，返回位序，无返回-1
8. `int IsIn(SeqList L, ListData x);` 判断x是否存在，存在返回1
9. `int NextElem(SeqList L, ListData x);` 获取x后继位序
10. `int PriorElem(SeqList L, ListData x);` 获取x前驱位序
11. `int ListInsert(SeqList &L, int i, ListData e);` 第i位插入e，成功1失败0
12. `int ListDelete(SeqList &L, int i);` 删除第i位元素，成功1失败0

## 三、基础操作函数
### 1. InitList(&L)：初始化线性表为空表
```c
int InitList(SeqList &L) {
    // 分配初始内存空间
    L.data = (ListData*)malloc(LIST_INIT_SIZE * sizeof(ListData));
    // 内存分配失败判断
    if (!L.data)
        exit(OVERFLOW);
    L.length = 0;               // 当前有效长度置0
    L.Listsize = LIST_INIT_SIZE;// 初始存储容量
    return OK;
}
```

### 2. DestroyList(&L)：回收顺序表全部内存，销毁结构体
```c
int DestroyList(SeqList &L) {
    free(L.data);       // 释放动态数组堆内存
    L.length = 0;       // 长度置0
    L.Listsize = 0;     // 容量置0
    return OK;
}
```
> 补充：C语言使用`malloc/free`管理堆内存；C++使用`new/delete`

### 3. ClearList(&L)：清空线性表（不释放内存，仅清除有效元素）
```c
int ClearList(SeqList &L) {
    L.length = 0;
    return OK;
}
```

### 4. ListEmpty(L)：判断线性表是否为空
为空返回1，非空返回0
写法1：
```c
int ListEmpty(SeqList L) {
    return !(L.length);
}
```
写法2：
```c
int ListEmpty(SeqList L) {
    if (L.length == 0) return 1;
    else return 0;
}
```

### 5. ListLength(L)：获取线性表当前有效长度
```c
int ListLength(SeqList L) {
    return L.length;
}
```

### 6. GetElem(L, i)：取表中第i个位置元素（位序1起始）
i 取值范围 `1 ≤ i ≤ L.length`，非法返回错误标识
```c
ListData GetElem(SeqList L, int i) {
    // 校验位序合法性 1<=i<=length
    if (i >= 1 && i <= L.length)
        return *(L.data + i - 1);
    else
        return NULL;
}
```

## 四、查找类函数
### 7-1. LocateElem(L, x)：查找元素x，返回位序（1-based），无则返回-1
while循环版本：
```c
int LocateElem(SeqList L, ListData x) {
    int i = 0;
    while ((i < L.length) && (*(L.data + i) != x))
        i++;
    if (i < L.length)
        return i + 1;  // 数组下标转逻辑位序:位序=下标+1
    else
        return -1;     // 查找失败返回-1
}
```
for循环版本：
```c
int LocateElem(SeqList L, ListData x) {
    for (int i = 0; i < L.length; i++)
        if (*(L.data + i) == x) return i + 1;
    return -1;
}
```

### 7-2. IsIn(L, x)：判断元素x是否存在于表中
存在返回1，不存在返回0
while循环版本：
```c
int IsIn(SeqList L, ListData x) {
    int i = 0, found = 0;
    while ((i < L.length) && !found) {
        if (*(L.data + i) != x)
            i++;
        else
            found = 1;
    }
    return found; // 存在返回1，不存在返回0
}
```
for循环版本：
```c
int IsIn(SeqList L, ListData x) {
    int found = 0;
    for (int i = 0; i < L.length; i++) {
        if (*(L.data + i) == x) {
            found = 1;
            break; // 找到直接跳出循环
        }
    }
    return found;
}
```

### 8. NextElem(L, x)：获取元素x的直接后继位序
找到x后有后继返回后继位序，无x/无后继返回-1
x下标\(i \in [0, L.length-1)\)->后继下标\(i+1\)
x位序\(i+1 \in [1, L.length)\)->后继位序\(i+2\)
```c
int NextElem(SeqList L, ListData x) {
    int i = 0;
    while ((i < L.length) && (*(L.data + i) != x))
        i++;
    // x存在且不是末尾元素，返回后继位序
    // if ((i >= 0) && (i < L.length - 1))
    if (i < L.length - 1) 
        return i + 2;//求位序
        // return i + 1;//求下标
    else
        return -1;
}
```

### 9. PriorElem(L, x)：获取元素x的直接前驱位序
找到x后有前驱返回前驱位序，无x/无前驱返回-1
x下标\(i \in [1, L.length)\)->前驱下标\(i - 1\)
x位序\(i+1 \in [2, L.length]\)->前驱位序\(i\)
```c
int PriorElem(SeqList L, ListData x) {
    int i = 0;
    while (i < L.length && *(L.data + i) != x)
        i++;
    // x存在且不是首元素，返回前驱位序
    if ((i > 0) && (i < L.length))
        return i;
    else
        return -1;
}
```

## 五、增删操作函数
### 10. ListInsert(&L, i, e)：在第i个元素前插入元素e
插入成功返回1，失败（i非法/内存扩容失败）返回0
```c
int ListInsert(SeqList &L, int i, ListData e) {
    ListData *newbase, *q, *p;
    // 位序合法性校验：i ∈ [1, L.length+1]
    if ((i < 1) || (i > L.length + 1))
        return ERROR;
    // 当前存储已满，执行扩容
    if (L.length == L.Listsize) {
        newbase = (ListData*)realloc(L.data,
            (L.Listsize + LISTINCREMENT) * sizeof(ListData));
        // realloc分配失败
        if (!newbase)
            exit(OVERFLOW);
        L.data = newbase;// 新基址
        L.Listsize += LISTINCREMENT; // 更新存储容量
    }
    //（0-based）(i-1) + p - q ∈ [i-1, L.length-1]
    q = L.data + i - 1; // 插入目标位置指针
    // 插入点后所有元素后移一位
    for (p = L.data + L.length - 1; p >= q; --p)
        *(p + 1) = *p;
    *q = e;       // 写入插入元素
    L.length++;   // 有效长度+1
    return OK;
}
```
> realloc作用：重新分配堆内存（原地或者异地），保留原有数据，扩容/缩容数组

### 11. ListDelete(&L, i)：删除第i个位置元素
删除成功返回1，失败（i非法）返回0
```c
int ListDelete(SeqList &L, int i) {
    ListData *p, *q;
    // 位序合法性校验：i ∈ [1, L.length]
    if (i < 1 || i > L.length)
        return ERROR;
    p = L.data + i - 1;          // 待删除元素指针
    q = L.data + L.length - 1;   // 表尾元素指针
    // 删除点后元素全部前移覆盖
    for (++p; p <= q; ++p)
        *(p - 1) = *p;
    L.length--;  // 有效长度-1
    return OK;
}
```
> 拓展写法：如需带出被删除元素，传引用参数
> `int ListDelete(SeqList &L, int i, ListData &e) { e = *p; ... }`

## 六、C 语言完整实现
```c
#include <stdio.h>
#include <stdlib.h>
#define LIST_INIT_SIZE 100
#define LISTINCREMENT 10
typedef int ListData;

// 动态分配顺序表结构体
typedef struct {
    ListData* data;
    int length;
    int Listsize;
} SeqList;

#define OK 1
#define ERROR 0
#define OVERFLOW -2

// 1. 初始化空表
int InitList(SeqList *L) {
    L->data = (ListData*)malloc(LIST_INIT_SIZE * sizeof(ListData));
    if (!L->data)
        exit(OVERFLOW);
    L->length = 0;
    L->Listsize = LIST_INIT_SIZE;
    return OK;
}

// 2. 销毁顺序表，释放堆内存
int DestroyList(SeqList *L) {
    free(L->data);
    L->length = 0;
    L->Listsize = 0;
    return OK;
}

// 3. 清空顺序表
int ClearList(SeqList *L) {
    L->length = 0;
    return OK;
}

// 4. 判断是否为空
int ListEmpty(SeqList L) {
    return !(L.length);
}

// 5. 获取表长度
int ListLength(SeqList L) {
    return L.length;
}

// 6. 获取第i个元素，成功返回1，元素存入e；失败返回0
int GetElem(SeqList L, int i, ListData *e) {
    if (i >= 1 && i <= L.length) {
        *e = *(L.data + i - 1);
        return OK;
    }
    return ERROR;
}

// 7-1 查找元素，返回位序，无则-1
int LocateElem(SeqList L, ListData x) {
    for (int i = 0; i < L.length; i++) {
        if (*(L.data + i) == x)
            return i + 1;
    }
    return -1;
}

// 7-2 判断元素是否存在
int IsIn(SeqList L, ListData x) {
    for (int i = 0; i < L.length; i++) {
        if (*(L.data + i) == x)
            return 1;
    }
    return 0;
}

// 8 获取x直接后继位序
int NextElem(SeqList L, ListData x) {
    int i = 0;
    while (i < L.length && *(L.data + i) != x)
        i++;
    if (i < L.length - 1)
        return i + 2;
    return -1;
}

// 9 获取x直接前驱位序
int PriorElem(SeqList L, ListData x) {
    int i = 0;
    while (i < L.length && *(L.data + i) != x)
        i++;
    if (i > 0 && i < L.length)
        return i;
    return -1;
}

// 10 插入元素
int ListInsert(SeqList *L, int i, ListData e) {
    ListData *newbase, *q, *p;
    if ((i < 1) || (i > L->length + 1))
        return ERROR;
    if (L->length == L->Listsize) {
        newbase = (ListData*)realloc(L->data,
            (L->Listsize + LISTINCREMENT) * sizeof(ListData));
        if (!newbase) exit(OVERFLOW);
        L->data = newbase;
        L->Listsize += LISTINCREMENT;
    }
    q = L->data + i - 1;
    for (p = L->data + L->length - 1; p >= q; --p)
        *(p + 1) = *p;
    *q = e;
    L->length++;
    return OK;
}

// 11 删除元素
int ListDelete(SeqList *L, int i) {
    ListData *p, *q;
    if (i < 1 || i > L->length)
        return ERROR;
    p = L->data + i - 1;
    q = L->data + L->length - 1;
    for (++p; p <= q; ++p)
        *(p - 1) = *p;
    L->length--;
    return OK;
}

// 辅助打印函数
void PrintList(SeqList L) {
    printf("动态顺序表(长度=%d,总容量=%d): ", L.length, L.Listsize);
    for (int i = 0; i < L.length; i++)
        printf("%d ", *(L.data + i));
    printf("\n");
}

// 测试主函数
int main() {
    SeqList L;
    ListData val;
    // 初始化
    InitList(&L);
    printf("【初始化】");
    PrintList(L);
    printf("是否为空：%s\n\n", ListEmpty(L) ? "是" : "否");

    // 插入测试
    ListInsert(&L, 1, 10);
    ListInsert(&L, 2, 20);
    ListInsert(&L, 3, 30);
    ListInsert(&L, 2, 15);
    printf("【插入10,20,30,第2位插入15】");
    PrintList(L);

    // 取值、查找、前驱后继
    GetElem(L, 3, &val);
    printf("第3位元素：%d\n", val);
    printf("20的位序：%d\n", LocateElem(L, 20));
    printf("99是否存在：%s\n", IsIn(L, 99) ? "存在" : "不存在");
    printf("20前驱位序：%d\n", PriorElem(L, 20));
    printf("20后继位序：%d\n\n", NextElem(L, 20));

    // 删除
    ListDelete(&L, 2);
    printf("【删除第2位】");
    PrintList(L);

    // 清空销毁
    ClearList(&L);
    printf("【清空表】");
    PrintList(L);
    DestroyList(&L);
    return 0;
}
```

## 八、符号访问规则
### 简要的类型定义：
```c
#define MAXSIZE 100 //最大长度
typedef struct {
    ElemType *elem;
    int length;
} SeqList;
```
### C++ 初始化表
### 1. 参数用引用
```cpp
Status InitList_Sq (SeqList &L) {
    L.elem = new ElemType[MAXSIZE];
    if (!L.elem) exit(OVERFLOW);
    L.length = 0;
    return OK;
}
```

### 2. 参数用指针
```cpp
Status InitList_Sq (SeqList *L) {
```
